# Install Prometheous stack

# Install via Helm
# Install via argocd = here I did via argocd


# Create prometheous application kube-prometheus-stack in argocd
--
apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: prometheus
  namespace: argocd
spec:
  project: default

  source:
    repoURL: https://prometheus-community.github.io/helm-charts
    chart: kube-prometheus-stack
    targetRevision: 84.4.0
    helm:
      values: |
        prometheus:
          service:
            type: NodePort
            nodePort: 30090

        grafana:
          service:
            type: NodePort
            nodePort: 30030

  destination:
    server: https://kubernetes.default.svc
    namespace: monitoring

  syncPolicy:
    automated:
      prune: true
      selfHeal: true
    syncOptions:
      - CreateNamespace=true

root@kube-master:~# cat > jenkins-pv.yml 
apiVersion: v1
kind: PersistentVolume
metadata:
  name: jenkins-pv
spec:
  capacity:
    storage: 10Gi
  accessModes:
    - ReadWriteMany
  persistentVolumeReclaimPolicy: Retain
  nfs:
    path: /nfs-share/jenkins-data
    server: kube-master

root@kube-master:~# cat > jenkins-pvc.yml 
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: jenkins
  namespace: jenkins
spec:
  accessModes:
    - ReadWriteMany
  resources:
    requests:
      storage: 10Gi
